CREATE TABLE MEMBER(
    MEMBER_ID NUMBER(13) PRIMARY KEY,                                    -- 전화번호
    MEMBER_PW VARCHAR2(30) NOT NULL,                                     -- 비밀번호
    MEMBER_NAME VARCHAR2(50) NOT NULL,                                   -- 회원명
    REGION_ID VARCHAR2(50) NOT NULL                                      -- 지역ID
)
;
